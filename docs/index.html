Beyond Latency: A Visual Guide to Live Subtitle Quality (6 Dimensions)

This notebook translates the Beyond Latency framework into a visual, non-technical guide. It is designed to be understandable for broadcast teams and useful for technical readers.

What you will learn

  • The six dimensions of live subtitle temporal quality
  • How to spot good vs bad in each dimension
  • How the dimensions trade off and combine in a composite score

Important measurement note (for these demo clips)

The paper defines latency as the delay from utterance end to subtitle appearance. The demo clips here use a synthetic base subtitle track; for consistency with the generated clips, we approximate "speech time" using each subtitle block start time. The conceptual definition remains utterance end.

# Helper functions

def parse_srt(path):
    # Parse SRT and return entries with start/end/text/wpm
    content = Path(path).read_text(encoding="utf-8")
    blocks = re.split(r"\n\s*\n", content.strip())
    entries = []

    def to_seconds(ts):
        h, m, s = ts.replace(",", ".").split(":")
        return int(h) * 3600 + int(m) * 60 + float(s)

    for block in blocks:
        lines = block.strip().split("\n")
        if len(lines) < 3:
            continue
        try:
            timing = lines[1]
            start_str, end_str = timing.split(" --> ")
            start = to_seconds(start_str)
            end = to_seconds(end_str)
            text = " ".join(lines[2:])
            duration = max(0.001, end - start)
            word_count = len(text.split())
            wpm = (word_count / duration) * 60
            entries.append({
                "start": start,
                "end": end,
                "text": text,
                "duration": duration,
                "word_count": word_count,
                "wpm": wpm,
            })
        except Exception:
            continue
    return entries


def write_srt(path, entries):
    def fmt(sec):
        h = int(sec // 3600)
        m = int((sec % 3600) // 60)
        s = sec % 60
        return f"{h:02d}:{m:02d}:{s:06.3f}".replace(".", ",")

    lines = []
    for i, e in enumerate(entries, 1):
        lines.append(str(i))
        lines.append(f"{fmt(e['start'])} --> {fmt(e['end'])}")
        lines.append(e["text"])
        lines.append("")
    path.write_text("\n".join(lines).strip() + "\n", encoding="utf-8")


def srt_to_vtt(srt_path, vtt_path):
    content = Path(srt_path).read_text(encoding="utf-8")
    content = content.replace(",", ".")
    vtt = "WEBVTT\n\n" + content
    vtt_path.write_text(vtt, encoding="utf-8")


def compute_latencies(base_entries, variant_entries, ref="start"):
    # Compute latency as variant_start - base_ref (ref: 'start' or 'end')
    base_ref = np.array([b[ref] for b in base_entries])
    variant_start = np.array([v["start"] for v in variant_entries])
    latencies = variant_start - base_ref
    return base_ref, variant_start, latencies


def compute_metrics(latencies):
    lat = np.asarray(latencies)
    jitter = float(np.mean(np.abs(np.diff(lat)))) if len(lat) > 1 else 0.0
    return {
        "mean": float(lat.mean()),
        "median": float(np.median(lat)),
        "std": float(lat.std()),
        "jitter": jitter,
        "l95": float(np.percentile(lat, 95)),
    }


def format_s(val):
    return f"{val:.2f}s"


def resolve_media_url(path):
    if USE_GITHUB_RAW:
        try:
            rel = path.resolve().relative_to(Path.cwd().resolve())
            return f"{GITHUB_RAW_BASE}/{rel.as_posix()}"
        except Exception:
            return f"{GITHUB_RAW_BASE}/{path.as_posix()}"
    return path.as_posix().replace(" ", "%20")


def resample_sequence(seq, n_out):
    if len(seq) == n_out:
        return seq.copy()
    x = np.linspace(0, len(seq) - 1, len(seq))
    xp = np.linspace(0, len(seq) - 1, n_out)
    return np.interp(xp, x, seq)


def ensure_variants_generated():
    # Generate variant SRTs for the input clip using reference distributions
    CLIPS_DIR.mkdir(parents=True, exist_ok=True)

    # Copy base SRT into output for downstream cells
    base_entries = parse_srt(BASE_SRT)
    base_out = CLIPS_DIR / "base_subtitles.srt"
    if not base_out.exists():
        write_srt(base_out, base_entries)

    # Load reference base
    ref_base = parse_srt(REFERENCE_BASE_SRT)
    ref_base_start = np.array([b["start"] for b in ref_base])

    dims = [
        "dimension_1_mean_latency",
        "dimension_2_consistency",
        "dimension_3_jitter",
        "dimension_4_robustness",
        "dimension_5_worst_case",
    ]

    base_start = np.array([b["start"] for b in base_entries])
    n_out = len(base_entries)

    for dim in dims:
        dim_out = CLIPS_DIR / dim
        dim_out.mkdir(parents=True, exist_ok=True)
        for variant in ["low", "high"]:
            ref_srt = REFERENCE_DIR / dim / f"{variant}_latency.srt"
            ref_entries = parse_srt(ref_srt)
            ref_lat = np.array([e["start"] for e in ref_entries]) - ref_base_start
            lat_out = resample_sequence(ref_lat, n_out)

            # Build new entries by applying latency to base timings
            updated = []
            for i, b in enumerate(base_entries):
                start = base_start[i] + lat_out[i]
                end = start + b["duration"]
                updated.append({"start": start, "end": end, "text": b["text"]})

            out_srt = dim_out / f"{variant}_latency.srt"
            write_srt(out_srt, updated)

            out_vtt = dim_out / f"{variant}_latency.vtt"
            srt_to_vtt(out_srt, out_vtt)

    # Also create VTT for base if needed
    base_vtt = CLIPS_DIR / "base_subtitles.vtt"
    if not base_vtt.exists():
        srt_to_vtt(base_out, base_vtt)

def plot_mean_latency(base_times, low_times, high_times, low_m, high_m):
    fig, ax = plt.subplots(1, 1, figsize=(10, 7))

    all_times = np.concatenate([base_times, low_times, high_times])
    t_min, t_max = all_times.min(), all_times.max()
    pad = (t_max - t_min) * 0.1

    ax.plot([t_min, t_max], [t_min, t_max], "k--", lw=2, alpha=0.5, label="Zero latency")

    ax.plot([t_min, t_max], [t_min + low_m['mean'], t_max + low_m['mean']],
            color="#2ca02c", lw=3, alpha=0.8, label=f"Low mean (mu={format_s(low_m['mean'])})")
    ax.plot([t_min, t_max], [t_min + high_m['mean'], t_max + high_m['mean']],
            color="#d62728", lw=3, alpha=0.8, label=f"High mean (mu={format_s(high_m['mean'])})")

    ax.scatter(base_times, low_times, color="#2ca02c", s=90, alpha=0.7, edgecolors="black", lw=1)
    ax.scatter(base_times, high_times, color="#d62728", s=90, alpha=0.7, edgecolors="black", lw=1)

    x_mid = (t_min + t_max) / 2
    ax.text(x_mid + pad * 0.2, x_mid + low_m['mean'], f"mean = {format_s(low_m['mean'])}",
            color="#2ca02c", weight="bold",
            bbox=dict(boxstyle="round,pad=0.3", facecolor="white", alpha=0.9))
    ax.text(x_mid + pad * 0.2, x_mid + high_m['mean'], f"mean = {format_s(high_m['mean'])}",
            color="#d62728", weight="bold",
            bbox=dict(boxstyle="round,pad=0.3", facecolor="white", alpha=0.9))

    ax.set_xlabel("Speech time (proxy)")
    ax.set_ylabel("Subtitle appearance time")
    ax.set_title("Mean Latency: average delay")
    ax.legend(loc="upper left", framealpha=0.95)
    ax.grid(alpha=0.3)
    ax.set_aspect("equal", adjustable="box")
    ax.set_xlim(t_min - pad, t_max + pad)
    ax.set_ylim(t_min - pad, t_max + pad)

    plt.tight_layout()
    plt.show()


def plot_consistency(lat_low, lat_high):
    fig, axes = plt.subplots(1, 2, figsize=(14, 5), sharey=True)

    def panel(ax, lat, color, title):
        idx = np.arange(len(lat))
        mean = lat.mean()
        std = lat.std()
        ax.plot(idx, lat, color=color, alpha=0.4)
        ax.scatter(idx, lat, color=color, s=70, edgecolors="black", lw=1)
        ax.axhline(mean, color=color, lw=2, alpha=0.8, label=f"Mean = {format_s(mean)}")
        ax.fill_between(idx, mean - std, mean + std, color=color, alpha=0.15,
                        label=f"+/-1sigma = {format_s(std)}")
        ax.text(0.02, 0.98, f"mean {format_s(mean)}\nsigma {format_s(std)}",
                transform=ax.transAxes, va="top",
                bbox=dict(boxstyle="round,pad=0.3", facecolor="white", alpha=0.9))
        ax.set_title(title)
        ax.set_xlabel("Subtitle index")
        ax.grid(alpha=0.3)
        ax.legend(loc="upper right", framealpha=0.95)

    panel(axes[0], lat_low, "#1f77b4", "High consistency (tight spread)")
    panel(axes[1], lat_high, "#ff7f0e", "Low consistency (wide spread)")
    axes[0].set_ylabel("Latency (s)")

    plt.tight_layout()
    plt.show()


def plot_jitter(lat_low, lat_high):
    fig, axes = plt.subplots(2, 2, figsize=(14, 8), sharex="col")

    max_lat = max(lat_low.max(), lat_high.max())
    max_jump = max(np.abs(np.diff(lat_low)).max(), np.abs(np.diff(lat_high)).max())

    def line_panel(ax, lat, color, title):
        idx = np.arange(len(lat))
        ax.plot(idx, lat, color=color, alpha=0.4)
        ax.scatter(idx, lat, color=color, s=70, edgecolors="black", lw=1)
        ax.set_title(title)
        ax.set_ylabel("Latency (s)")
        ax.set_ylim(0, max_lat * 1.15)
        ax.grid(alpha=0.3)

    def bar_panel(ax, lat, color, title):
        jumps = np.abs(np.diff(lat))
        idx = np.arange(1, len(lat))
        ax.bar(idx, jumps, color=color, alpha=0.7, edgecolor="black")
        mean_jump = jumps.mean() if len(jumps) else 0
        ax.axhline(mean_jump, color="black", lw=2, linestyle=":",
                   label=f"Avg jump = {format_s(mean_jump)}")
        ax.text(0.02, 0.98, f"mean jump {format_s(mean_jump)}",
                transform=ax.transAxes, va="top",
                bbox=dict(boxstyle="round,pad=0.3", facecolor="white", alpha=0.9))
        ax.set_title(title)
        ax.set_xlabel("Between subtitle i and i+1")
        ax.set_ylim(0, max_jump * 1.15)
        ax.grid(alpha=0.3)
        ax.legend(loc="upper right", framealpha=0.95)

    line_panel(axes[0, 0], lat_low, "#2ca02c", "Low jitter: latency by subtitle order")
    line_panel(axes[0, 1], lat_high, "#9467bd", "High jitter: latency by subtitle order")

    bar_panel(axes[1, 0], lat_low, "#2ca02c", "Low jitter: consecutive jumps")
    bar_panel(axes[1, 1], lat_high, "#9467bd", "High jitter: consecutive jumps")

    axes[1, 0].set_ylabel("Consecutive change in latency (s)")

    plt.tight_layout()
    plt.show()

def categorize_wpm(wpm):
    if wpm < 100:
        return "slow"
    if wpm < 160:
        return "normal"
    if wpm < 200:
        return "elevated"
    return "fast"


def plot_robustness(base_entries, lat_low, lat_high):
    # 2x2 layout: top row high robustness, bottom row low robustness
    times = np.array([b["start"] for b in base_entries])
    wpm = np.array([b["wpm"] for b in base_entries])
    categories = np.array([categorize_wpm(v) for v in wpm])

    bands = ["normal", "elevated", "fast"]
    band_labels = ["Normal (100-160)", "Elevated (160-200)", "Fast (>200)"]
    band_colors = ["#1f77b4", "#ff7f0e", "#d62728"]

    def band_means(lat):
        means = []
        for b in bands:
            mask = categories == b
            means.append(lat[mask].mean() if mask.any() else np.nan)
        return np.array(means)

    high_means = band_means(lat_high)
    low_means = band_means(lat_low)
    max_bar = np.nanmax([np.nanmax(high_means), np.nanmax(low_means)])
    max_time = np.max([lat_high.max(), lat_low.max()])

    def bar_panel(ax, lat, means, title):
        x = np.arange(len(bands))
        ax.bar(x, means, color=band_colors, edgecolor="black")
        for i, v in enumerate(means):
            if np.isfinite(v):
                ax.text(x[i], v + 0.05, format_s(v), ha="center", va="bottom", fontsize=9)
        ax.set_xticks(x)
        ax.set_xticklabels(band_labels)
        ax.set_ylabel("Mean latency (s)")
        ax.set_title(title)
        ax.grid(axis="y", alpha=0.3)
        ax.set_ylim(0, max_bar * 1.15)

        normal_mask = categories == "normal"
        fast_mask = categories == "fast"
        if normal_mask.any() and fast_mask.any():
            rs = means[0] / means[2] if np.isfinite(means[2]) else float("nan")
            ax.text(0.98, 0.98, f"Rs: {rs:.2f}",
                    transform=ax.transAxes, ha="right", va="top",
                    bbox=dict(boxstyle="round,pad=0.3", facecolor="white", alpha=0.9))

    def time_panel(ax, lat, title):
        for band, color in zip(bands, band_colors):
            mask = categories == band
            if mask.any():
                ax.scatter(times[mask], lat[mask], color=color, s=80,
                           edgecolors="black", lw=1, label=band)
        ax.plot(times, lat, color="#555", alpha=0.3, lw=1)
        ax.set_title(title)
        ax.set_xlabel("Time (s)")
        ax.set_ylabel("Latency (s)")
        ax.grid(alpha=0.3)
        ax.set_ylim(0, max_time * 1.15)
        ax.legend(loc="upper right", framealpha=0.95)

    fig, axes = plt.subplots(2, 2, figsize=(14, 8))

    bar_panel(axes[0, 0], lat_high, high_means, "High robustness: latency by band")
    time_panel(axes[0, 1], lat_high, "High robustness: time vs delay")

    bar_panel(axes[1, 0], lat_low, low_means, "Low robustness: latency by band")
    time_panel(axes[1, 1], lat_low, "Low robustness: time vs delay")

    plt.tight_layout()
    plt.show()

def plot_worst_case(lat_low, lat_high):
    fig, axes = plt.subplots(1, 2, figsize=(14, 5), sharey=True)

    def panel(ax, lat, color, title):
        sorted_lat = np.sort(lat)
        pct = np.linspace(0, 100, len(sorted_lat))
        l95 = np.percentile(lat, 95)
        mean = lat.mean()
        ax.plot(pct, sorted_lat, color=color, lw=2)
        ax.axvline(95, color=color, lw=2, linestyle="--")
        ax.axhline(l95, color=color, lw=2, linestyle=":")
        ax.axhline(mean, color=color, lw=1.5, linestyle="--", alpha=0.7)
        ax.set_title(title)
        ax.set_xlabel("Percentile")
        ax.grid(alpha=0.3)
        ax.text(0.02, 0.95, f"L95 = {format_s(l95)}", transform=ax.transAxes,
                va="top", bbox=dict(boxstyle="round,pad=0.4", facecolor="white", alpha=0.9))
        ax.text(0.02, 0.85, f"mean = {format_s(mean)}", transform=ax.transAxes,
                va="top", bbox=dict(boxstyle="round,pad=0.4", facecolor="white", alpha=0.9))

    panel(axes[0], lat_low, "#2ca02c", "Low worst-case (tight tail)")
    panel(axes[1], lat_high, "#d62728", "High worst-case (long tail)")
    axes[0].set_ylabel("Latency (s)")

    plt.show()

Dimension 3: Jitter (JL)

Jitter captures the moment-to-moment jumps in latency. Even if average and spread look similar, jitter can make the experience feel jerky.

Think of this as: Do subtitles arrive in a smooth rhythm, or do they jump around?

Dimension 4: Robustness (Rs)

Robustness measures how well timing holds up when speech speeds up.

Speech rate bands (from the paper):

  • Normal: 100-160 WPM
  • Elevated: 160-200 WPM
  • Fast: >200 WPM

Dimension 5: Degradation Mode (D)

Degradation mode describes how a system fails under stress:

  • Latency degradation: accuracy stays good but subtitles arrive late
  • Accuracy degradation: timing stays good but errors increase
  • Both degrade: late and wrong

This dimension is categorical, not numeric, and is often the easiest for stakeholders to describe.

Dimension 6: Worst-Case Latency (L95)

Worst-case (L95) is the 95th percentile of latency. It captures the "tail" -- the rare but painful moments.

Think of this as: How bad does it get when things go wrong?

Putting it all together

The Subtitle Temporal Quality Score (STQS) combines the numeric dimensions into one score (weights from the paper):

STQS = 0.25*f(muL) + 0.30*g(sigmaL) + 0.15*h(JL) + 0.15*Rs + 0.15*k(L95)

Note: Degradation Mode (D) is categorical, so it is not folded into the numeric score.

In[14]:
# Summary table (computed from demo clips)

def summarize_dimension(dim_key, label, good_variant="low", bad_variant="high"):
    base = parse_srt(CLIPS_DIR / "base_subtitles.srt")
    good = parse_srt(CLIPS_DIR / dim_key / f"{good_variant}_latency.srt")
    bad = parse_srt(CLIPS_DIR / dim_key / f"{bad_variant}_latency.srt")

    _, _, lat_good = compute_latencies(base, good, ref="start")
    _, _, lat_bad = compute_latencies(base, bad, ref="start")

    m_good = compute_metrics(lat_good)
    m_bad = compute_metrics(lat_bad)

    return {
        "Dimension": label,
        "Good (key metric)": None,
        "Bad (key metric)": None,
        "Notes": "",
    }, m_good, m_bad
